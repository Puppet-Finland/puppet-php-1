  Array[String, 1]     $extensions,
  Optional[String]     $package_name = undef,
  Hash[String, String] $settings     = {},
) {

  include ::php::extension::pdo

  ::php::extension { 'mysql':

    package_name => $package_name,
    settings     => $settings,

  }
